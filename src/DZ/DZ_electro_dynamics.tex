\par \textbf{Решение}:

%\par
\par Матричный элемент:
\[
    A = \sqrt{\frac{2\pi c^2 \hslash}{V \omega n^2}}
\]
\[
    M_{cv} = \frac{e}{m_0 c} A p_{cv} = \frac{ep_{cv}}{m_0 c} \sqrt{\frac{2\pi c^2 \hslash}{V \omega n^2}}
\]
\par Скорость переходов:
\[
    W = \frac{2\pi}{\hslash} \sum_k \left| M_{cv} \right|^2 \delta \left( E_g + \frac{\hslash^2 k^2}{2\mu} -\hslash \omega \right) =
\]
\[
    = \frac{2\pi}{\hslash} \frac{V}{(2\pi)^3} \int 4\pi k^2 dk \frac{e^2 p^2_{cv}}{m_0^2 c^2} \frac{2\pi c^2 \hslash}{V \omega n^2}  \delta \left( E_g + \frac{\hslash^2 k^2}{2\mu} -\hslash \omega \right) =
\]
\[
    = \frac{e^2 p_{cv}^2}{m_0^2 \omega n^2} \left( \frac{2\mu}{\hslash^2} \right)^{3/2} \sqrt{\hslash \omega - E_g}
\]
\par Коэфф поглощения:
\[
    \alpha_0 = W \frac{n}{c} = \frac{e^2 p_{cv}^2}{m_0^2 \omega c n} \left( \frac{2\mu}{\hslash^2} \right)^{3/2} \sqrt{\hslash \omega - E_g}
\]
\[
    \alpha = \alpha_0 f = W \frac{n}{c} = \frac{e^2 p_{cv}^2}{m_0^2 \omega c n} \left( \frac{2\mu}{\hslash^2} \right)^{3/2} \sqrt{\hslash \omega - E_g} \pi \frac{z e^z}{\sinh(z)}
\]
\par $\hslash \omega \rightarrow E_g \Rightarrow f \rightarrow 2z $:
\[
    \alpha = \frac{2\pi e^2 p_{cv}^2}{\hslash^3 c m_0^2 \omega n } (2\mu)^{3/2} \sqrt{E_b}
\]
\par Подставив $ E_b = \frac{e^2}{2n^2 a_B^2}$; $ \omega = E_g/\hslash$; $a_B = \frac{\hslash^2 n^2}{\mu e^2}$:
\[
    \alpha = \frac{4\pi r_{cv}^2 \hslash n}{c E_g m_0^2 a_B^2}
\]
