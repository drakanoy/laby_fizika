\par Условие:
\par Решить уравнение на матрицу плотности для любого поля и найти коэфф затухания.
\par

\par \textbf{Решение}:
%\par
%\includegraphics[width=1\textwidth]{photo_11.jpg}
%\par
\par Уравнение на матрицу плотности:
\[
    \frac{d \rho}{dt} = \frac{-i}{\hslash} \left[ H, \rho \right] + L(\rho)
\]
\par Учитывая $\rho_{ge}^{*} = \rho_{eg}$; $\rho_{ee} + \rho_{gg} = 1$, получим 2 линейно независимых уравнения на компоненты матрицы плотности
\begin{eqnarray*}
    \begin{cases}
        \frac{d \rho_{ee}}{dt} = i U^* e^{i\omega t} \rho_{eg} - i U e^{-i\omega t} \rho_{ge} - \frac{\rho_{ee}}{T_1} \\
        \frac{d \rho_{eg}}{dt} = -i\omega_0 \rho_{eg} - i U e^{-i\omega t} \left( 1 - 2 \rho_{ee} \right) - \frac{\rho_{eg}}{T_2}
    \end{cases}
\end{eqnarray*}
\par Хотим стационарные решения при $t \rightarrow \infty$:
\[
    \frac{d \rho_{ee}}{dt} = 0 \,\,\, ; \,\,\, \frac{d \rho_{eg}}{dt} = -i \omega \rho_{eg}
\]
\par $\Gamma := \frac{1}{T_2}$, подставив все в систему - получаем:
\[
    \rho_{eg} = \frac{-i U e^{-i\omega t} \left( 1 - 2 \rho_{ee} \right)}{i\left( \omega_0 - \omega \right) + \Gamma}
\]
\[
    \frac{\rho_{ee}}{T_1} = \left| U \right|^2 \left( 1 - 2 \rho_{ee} \right) \frac{2\Gamma}{\left( \omega_0 - \omega \right)^2 + \Gamma^2}
\]
\par Тогда:
\[
    \rho_{ee} = \frac{2T_1 \Gamma \left| U \right|^2}{\left( \omega_0 - \omega \right)^2 + \Gamma^2 + 4T_1 \Gamma \left| U \right|^2}
\]
\[
    \rho_{eg} = \frac{-i U e^{-i\omega t}}{i\left( \omega_0 - \omega \right) + \Gamma} \left( \frac{\left( \omega_0 - \omega \right)^2 + \Gamma^2 }{\left( \omega_0 - \omega \right)^2 + \Gamma^2 + 4T_1 \Gamma \left| U \right|^2} \right)
\]
\[
    \rho_{ge}^{*} = \rho_{eg}
\]
\[
    \rho_{gg} = 1 - \rho_{ee}
\]
\par Теперь найдем коэфф затухания:
\[
    \alpha = \frac{\rho_{ee}}{T_1} \frac{N_{QD}}{N} \frac{n}{c}
\]
\par Учитывая $\Gamma = \frac{1}{2\tau_0}$ и $\omega = \omega_0 $:
\[
    \alpha = \frac{4 \left| U \right|^2 \tau_0 }{1 + 8 T_1 \left| U \right|^2 \tau_0} \frac{N_{QD}}{N} \frac{n}{c}
\]
\par Подставляя $\left| U \right|^2 \tau_0 = \frac{3c A_0^2}{4n \omega_0 \hslash} $ и $ N = \frac{\omega_0 n^2 A_0^2}{2\pi \hslash c^2} $, получаем:
\[
    \alpha = \frac{6\pi N_{QD} \frac{c^2}{\omega_0^2 n^2}}{1+6T_1 \frac{c A_0^2}{\omega_0 n \hslash }}
\]
\par Введем $q := \frac{n\omega_0}{c}$
\[
    \alpha = \frac{6\pi N_{QD} \frac{1}{q^2}}{1+6T_1 \frac{A_0^2}{q \hslash }}
\]

\[
    \rho_{ee} = \frac{2T_1 \Gamma \left| U \right|^2}{\left( \omega_0 - \omega \right)^2 + \Gamma^2 + 4T_1 \Gamma \left| U \right|^2}
\]
\[
    \rho_{eg} = \frac{-i U e^{-i\omega t}}{i\left( \omega_0 - \omega \right) + \Gamma} \left( \frac{\left( \omega_0 - \omega \right)^2 + \Gamma^2 }{\left( \omega_0 - \omega \right)^2 + \Gamma^2 + 4T_1 \Gamma \left| U \right|^2} \right)
